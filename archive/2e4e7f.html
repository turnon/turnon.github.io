_.bind与_.bindAll

2016-04-09 javascript underscore
最简单的自定义bind：


function bind(func, ctx){
  return function(){
    return func.apply(ctx, arguments);
  }
}


underscore的_.bind：


var FuncProto = Function.prototype;
var nativeBind = FuncProto.bind;

_.bind = function(func, context) {

  // 如果支持ES5的bind，则使用之
  if (nativeBind && func.bind === nativeBind) return nativeBind.apply(func, slice.call(arguments, 1));

  if (!_.isFunction(func)) throw new TypeError('Bind must be called on a function');

  // 支持curry
  var args = slice.call(arguments, 2);
  var bound = function() {
    return executeBound(func, bound, context, this, args.concat(slice.call(arguments)));
  };
  return bound;
};

var executeBound = function(sourceFunc, boundFunc, context, callingContext, args) {

  // 如果this不是指向绑定后的函数自身（即非new操作），则如同最简单的bind
  if (!(callingContext instanceof boundFunc)) return sourceFunc.apply(context, args);

  // 如果是new操作，则手动将this指回原函数构建出的对象
  var self = baseCreate(sourceFunc.prototype);
  var result = sourceFunc.apply(self, args);

  // 如果原函数有return语句，则返回该return，否则返回new出的对象
  if (_.isObject(result)) return result;
  return self;
};


_.bindAll用于将某对象的方法绑定到该对象上，使之不受call，apply影响。当该对象有function(){callbacks.forEach(function(f){f.apply(this)})}之类需要绑定原对象来执行的callback时，相当有用：


_.bindAll = function(obj) {
  var i, length = arguments.length, key;
  if (length <= 1) throw new Error('bindAll must be passed function names');
  for (i = 1; i < length; i++) {
    key = arguments[i];
    obj[key] = _.bind(obj[key], obj);
  }
  return obj;
};

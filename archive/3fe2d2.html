js module pattern惯用法

2016-08-10 javascript pattern
Basic Module Pattern

目的：

Private members live in the closure.
Public members are exposed in the return object.

要求：

There must be an outer enclosing function, and it must be invoked at least once (each time creates a new module instance).
The enclosing function must return back at least one inner function, so that this inner function has closure over the private scope, and can access and/or modify that private state.

一般地，就是返回一个字面量对象，使其引用到刚才执行的函数所产生的新的作用域（以该作用域为父作用域），但具体做法主要有三种

Revealing Module Pattern（最常用）

arranges all variables and methods at the top of the IIFE  and simply assigns them to the returned object

var welcomeModule = (function(){
  // 无论public还是private，都定义在返回对象之外
  var name = "John";
  var hello = function(){ console.log("Hello, " + name + "!");}
  var welcome = function() { hello(); console.log("Welcome to StackOverflow!");}
  return {
    name: name,
    sayHello: hello, // 增加间接层sayHello -> hello
    sayWelcome: welcome
  }
})();


当你想将name和sayHello隐藏时，只需将其注释掉，这对代码结构不会有很大影响。

var welcomeModule = (function(){
  var name = "John";
  var hello = function(){ console.log("Hello, " + name + "!");}
  var welcome = function() { hello(); console.log("Welcome to StackOverflow!");}
  return {
    //name: name,
    //sayHello: hello,
    sayWelcome: welcome
  }
})();


但它也有缺点，当你想以patch方式重定义module中某个方法/属性时，如果该方法/属性有被其它方法引用，则其它方法并不会受到改变，因为“其它方法”引用的还是closure中原来的方法/属性：

welcomeModule.sayHello = function(){ console.log('Are you ok ?');};
welcomeModule.sayWelcome(); // => Hello, John Welcome to StackOverflow


Module Pattern with Object Literal（最古老）

var welcomeModule = (function(){
  return {
    name: "John",
    // 因为私有的也写在返回对象中，所以对其的引用必须带上this
    sayHello: function(){ console.log("Hello, " + this.name + "!");},
    welcome: function() { this.sayHello(); console.log("Welcome to StackOverflow!");}
  }
})();


如果想将某些成员私有化，则引用也许改变

var welcomeModule = (function(){
  var name: "John";
  var sayHello = function(){ console.log("Hello, " + name + "!");};
  return {
    //name: "John",
    //sayHello: function(){ console.log("Hello, " + this.name + "!");},
    // 去掉了this，以改用closure的
    welcome: function() { sayHello(); console.log("Welcome to StackOverflow!");}
  }
})();


Module Pattern with Return Object Stub

先构造一个空的对象，再填充成员进去，最后返回它

var welcomeModule = (function(){
  var stub = {};
  stub.name = "John";
  stub.sayHello = function(){ console.log("Hello, " + stub.name + "!");},
  stub.sayWelcome: function() { stub.sayHello(); console.log("Welcome to StackOverflow!");}
  return stub;
})();


如果需要调整private/public，也需要大动筋骨，包括修改this引用：

var welcomeModule = (function(){
  var stub = {};
  var name = "John";
  var sayHello = function(){ console.log("Hello, " + name + "!");}

  stub.sayWelcome: function() { sayHello(); console.log("Welcome to StackOverflow!");}
  return stub;
})();


Loose augmentation of modules

在异步加载（无法确定加载顺序）时使用，做法是，每个augmentation IIFE传入“模块名或空对象”，返回模块

某js文件

var GlobalModule = (function(module){

  var private_data = 123;

  module.method_1 = function(arg){
    // deal with arg and private_data
  };

  return module;

})(GlobalModule || {});


另一js文件

var GlobalModule = (function(module){

  var private_data = 456;

  module.method_2 = function(arg){
    // deal with arg and private_data
  };

  return module;

})(GlobalModule || {});


Private Members for Constructors

一般做法：

function Person(name) {
  var age = 25;
  this.name = name;
  this.getAge = function() {
    return age;
  };
  this.growOlder = function() {
    age++;
  };
}


避免重复生成public方法：

var Person = (function() {
  var age = 25;
  function InnerPerson(name) {
    this.name = name;
  }
  InnerPerson.prototype.getAge = function() {
    return age;
  };
  InnerPerson.prototype.growOlder = function() {
    age++;
  };
  return InnerPerson;
}());

decorator pattern

2015-04-05 javascript pattern underscore
装饰者与被装饰者实现同样的接口
装饰者持有被装饰者的实例
在调用被装饰者的方法时，加插功能

将jQuery.extend

对于functional js来说，所有function都具备的call和apply本来就相当于“同样接口”，所以只需用function包装function即可

见underscore中的两种实现：


_.wrap = function(func, wrapper) {
  // 让wrapper接收func，wrapper内部会使用它
  return _.partial(wrapper, func);
};



_.compose = function() {
  var args = arguments;
  var start = args.length - 1;
  return function() {
    var i = start;
    // 从最后的function开始（当然改成从头开始也可以）
    var result = args[start].apply(this, arguments);
    while (i--) result = args[i].call(this, result);
    return result;
  };
};



// 当然改成从头开始也可以
function compose() {
  var args = arguments;
  var end = args.length - 1;
  return function() {
    var i = 0;
    var result = args[i].apply(this, arguments);
    while (i++ < end) result = args[i].call(this, result);
    return result;
  };
};
